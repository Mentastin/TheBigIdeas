\chapter{Circuits and circuit topologies}
\begin{definition}
A \udef{circuit} is a symmetric digraph $\sSet{C,N}$.
\begin{itemize}
\item The vertices in $C$ are called \udef{components}.
\item The edges in $N$ are called \udef{nets}.
\end{itemize}

A \udef{complex network} is a circuit $\sSet{C, N}$ together with a pair of functions
\begin{itemize}
\item $V: N \to \C$ such that $V(n) = - V(n^t)$ for all $n\in N$;
\item $I: N \to \C$ such that $I(n) = - I(n^t)$ for all $n\in N$.
\end{itemize} 
A \udef{time-dependent network} is a circuit $\sSet{C, N}$ together with a pair of functions
\begin{itemize}
\item $V: N \to (\R\to \R)$ such that $V(n) = - V(n^t)$ for all $n\in N$;
\item $I: N \to (\R\to \R)$ such that $I(n) = - I(n^t)$ for all $n\in N$.
\end{itemize} 
\end{definition}
